3. Callbacks

⚠️ Important: Open the browser console to see the results

Exercise 1

Function:

                    
                        const procesar = (numero, callback) => {
                            callback(numero);
                        };
                        
                        // Ejemplo de función callback
                        const duplicar = (num) => {
                            console.log(`El doble de ${num} es ${num * 2}`);
                        };

Example of usage:

                    
                        procesar(5, duplicar);
                        // El doble de 5 es 10
                        
                        procesar(8, (num) => {
                            console.log(`El cuadrado de ${num} es ${num * num}`);
                        });
                        // El cuadrado de 8 es 64

Exercise 2

Function:

                    
                        const calculadora = (num1, num2, callback) => {
                            return callback(num1, num2);
                        };
                        
                        // Función callback para suma
                        const sumar = (a, b) => {
                            const resultado = a + b;
                            console.log(`${a} + ${b} = ${resultado}`);
                            return resultado;
                        };

Example of usage:

                    
                        calculadora(5, 3, sumar);
                        // 5 + 3 = 8
                        
                        calculadora(10, 7, (a, b) => {
                            const resultado = a - b;
                            console.log(`${a} - ${b} = ${resultado}`);
                            return resultado;
                        });
                        // 10 - 7 = 3

Exercise 3

Function:

                    
                        const esperarISaludar = (nombre, callback) => {
                            setTimeout(() => {
                                callback(nombre);
                            }, 2000);
                        };
                        
                        // Función callback para saludar
                        const saludarPersona = (nombre) => {
                            console.log(`¡Hola, ${nombre}! Han pasado 2 segundos.`);
                        };

Example of usage:

                    
                        esperarISaludar('Ana', saludarPersona);
                        // (Después de 2 segundos) ¡Hola, Ana! Han pasado 2 segundos.
                        
                        esperarISaludar('Carlos', (nombre) => {
                            console.log(`Bienvenido, ${nombre}!`);
                        });
                        // (Después de 2 segundos) Bienvenido, Carlos!

Exercise 4

Function:

                    
                        const procesarElements = (array, callback) => {
                            for (let i = 0; i < array.length; i++) {
                                callback(array[i], i);
                            }
                        };
                        
                        // Función callback de ejemplo
                        const mostrarElemento = (elemento, indice) => {
                            console.log(`Índice ${indice}: ${elemento}`);
                        };

Example of usage:

                    
                        const frutas = ['manzana', 'banana', 'naranja'];
                        procesarElements(frutas, mostrarElemento);
                        // Índice 0: manzana
                        // Índice 1: banana
                        // Índice 2: naranja
                        
                        procesarElements([1, 2, 3], (num) => {
                            console.log(`El doble de ${num} es ${num * 2}`);
                        });
                        // El doble de 1 es 2
                        // El doble de 2 es 4
                        // El doble de 3 es 6

Exercise 5

Function:

                    
                        const procesarCadena = (cadena, callback) => {
                            const cadenaTransformada = cadena.toUpperCase();
                            callback(cadenaTransformada);
                        };
                        
                        // Función callback de ejemplo
                        const mostrarCadena = (cadena) => {
                            console.log(`Cadena transformada: ${cadena}`);
                        };

Example of usage:

                    
                        procesarCadena('hola mundo', mostrarCadena);
                        // Cadena transformada: HOLA MUNDO
                        
                        procesarCadena('javascript', (cadena) => {
                            console.log(`La cadena "${cadena}" tiene ${cadena.length} caracteres`);
                        });
                        // La cadena "JAVASCRIPT" tiene 10 caracteres
