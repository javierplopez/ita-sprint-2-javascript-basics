5. Array Transformations

⚠️ Important: Open the browser console to see the results

Exercise 1

Code:

                    
                        const numeros = [1, 2, 3, 4];
                        const cuadrados = numeros.map(num => num * num);

Example of usage:

                    
                        console.log('Array original:', numeros);
                        console.log('Array cuadrados:', cuadrados);

Exercise 2

Code:

                    
                        const numeros = [1, 2, 3, 4];
                        const pares = numeros.filter(num => num % 2 === 0);

Example of usage:

                    
                        console.log('Array original:', numeros);
                        console.log('Números pares:', pares);

Exercise 3

Code:

                    
                        const numeros = [1, 10, 8, 11];
                        const mayorA10 = numeros.find(num => num > 10);

Example of usage:

                    
                        console.log('Array original:', numeros);
                        console.log('Primer número mayor a 10:', mayorA10);

Exercise 4

Code:

                    
                        const numeros = [13, 7, 8, 21];
                        const sumaTotal = numeros.reduce((acumulador, num) => acumulador + num, 0);

Example of usage:

                    
                        console.log('Array original:', numeros);
                        console.log('Suma total:', sumaTotal);

Exercise 5

Code:

                    
                        const numeros = [1, 3, 7, 10, 15, 17, 11, 5, 8, 12, 9];
                        const resultado = numeros.filter(num => num >= 10).map(num => num * 2).reduce((acc, num) => acc + num, 0);

Example of usage:

                    
                        console.log('Array original:', numeros);
                        console.log('Resultado:', resultado);

Exercise 6

Code:

                    
                        const numeros = [11, 12, 13, 14];
                        const todosmayor10 = numeros.every(num => num > 10);
                        const algunomayor10 = numeros.some(num => num > 10);

Example of usage:

                    
                        console.log('Array original:', numeros);
                        console.log('Todos mayores que 10:', todosmayor10);
                        console.log('Algunos mayores que 10:', algunomayor10);
