1. Arrow Functions

⚠️ Important: Open the browser console to see the results

Exercise 1

Function:

                    
                        const add = (a, b) => a + b;

Example of usage:

                    
                        const result = add(2, 3);
                        console.log(result); // 5

Exercise 2

Function:

                    
                        const randomNumber = () => Math.floor(Math.random() * 101);

Example of usage:

                    
                        const number = randomNumber();
                        console.log(number); // Random number between 0 and 100

Exercise 3

Function:

                    
                        class Person {
                            constructor(name) {
                                this.name = name;
                            }
                            
                            greet = () => {
                                console.log(`Hola, ${this.name}`);
                            }
                        }

Example of usage:

                    
                        const person = new Person('Alice');
                        person.greet(); // Hola, Alice

Exercise 4

Function:

                    
                        const printNumbers = (numbers) => {
                            for (let i = 0; i < numbers.length; i++) {
                                const printNumber = () => console.log(numbers[i]);
                                printNumber();
                            }
                        };

Example of usage:

                    
                        const numbersArray = [1, 2, 3, 4, 5];
                        printNumbers(numbersArray); // 1, 2, 3, 4, 5 (each on new line)

Exercise 5

Function:

                    
                        const delayedMessage = () => {
                            setTimeout(() => {
                                console.log('¡Mensaje después de 3 segundos!');
                            }, 3000);
                        };

Example of usage:

                    
                        delayedMessage(); // Waits 3 seconds, then prints: ¡Mensaje después de 3 segundos!
