4. Rest and Spread Operators

⚠️ Important: Open the browser console to see the results

Exercise 1

Code:

                    
                        const firstNumbers = [1, 2, 3];
                        const secondNumbers = [4, 5, 6];
                        
                        const combinedNumbers = [...firstNumbers, ...secondNumbers];

Example:

                    
                        console.log('First numbers:', firstNumbers);
                        console.log('Second numbers:', secondNumbers);
                        console.log('Combined numbers:', combinedNumbers);

Exercise 2

Code:

                    
                        const sum = (...numbers) => {
                            return numbers.reduce((total, num) => total + num, 0);
                        };

Example:

                    
                        console.log(sum(1, 2, 3));
                        console.log(sum(10, 20, 30, 40));

Exercise 3

Code:

                    
                        const object1 = {
                            name: 'Ana',
                            age: 25,
                            city: 'Madrid'
                        };
                        
                        const object2 = { ...object1 };
                        
                        object2.age = 30;
                        object2.profession = 'Developer';

Example:

                    
                        console.log('Object 1:', object1);
                        console.log('Object 2:', object2);
                        console.log('Are they the same object?', object1 === object2);

Exercise 4

Code:

                    
                        const colors = ['red', 'green', 'blue', 'yellow', 'orange', 'purple'];
                        
                        const [first, second, ...rest] = colors;

Example:

                    
                        console.log('First element:', first);
                        console.log('Second element:', second);
                        console.log('Rest of elements:', rest);

Exercise 5

Code:

                    
                        const greet = (name, surname, age) => {
                            return `Hello, ${name} ${surname}. You are ${age} years old.`;
                        };
                        
                        const personData = ['Ana', 'García', 28];

Example:

                    
                        console.log(greet(...personData));

Exercise 6

Code:

                    
                        const objectA = {

                        const objectB = {
