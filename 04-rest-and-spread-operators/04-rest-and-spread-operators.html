4. Rest and Spread Operators

⚠️ Important: Open the browser console to see the results

Exercise 1

Code:

                    
                        const array1 = [1, 2, 3];
                        const array2 = [4, 5, 6];
                        
                        // Usando operador spread para combinar arrays
                        const array3 = [...array1, ...array2];

Example of usage:

                    
                        console.log('Array 1:', array1); // [1, 2, 3]
                        console.log('Array 2:', array2); // [4, 5, 6]
                        console.log('Array 3 (combinado):', array3); // [1, 2, 3, 4, 5, 6]
                        
                        // También se puede insertar elementos en el medio
                        const array4 = [...array1, 'intermedio', ...array2];
                        console.log('Array 4:', array4); // [1, 2, 3, 'intermedio', 4, 5, 6]

Exercise 2

Function:

                    
                        const suma = (...numeros) => {
                            return numeros.reduce((total, num) => total + num, 0);
                        };

Example of usage:

                    
                        console.log(suma(1, 2, 3)); // 6
                        console.log(suma(10, 20, 30, 40)); // 100
                        console.log(suma(5)); // 5
                        console.log(suma()); // 0
                        
                        // También funciona con arrays usando spread
                        const numerosArray = [7, 8, 9];
                        console.log(suma(...numerosArray)); // 24

Exercise 3

Code:

                    
                        const objeto1 = {
                            nombre: 'Ana',
                            edad: 25,
                            ciudad: 'Madrid'
                        };
                        
                        // Copiando objeto con spread
                        const objeto2 = { ...objeto1 };
                        
                        // Modificando objeto2
                        objeto2.edad = 30;
                        objeto2.profesion = 'Desarrolladora';

Example of usage:

                    
                        console.log('Objeto1:', objeto1);
                        // { nombre: 'Ana', edad: 25, ciudad: 'Madrid' }
                        
                        console.log('Objeto2:', objeto2);
                        // { nombre: 'Ana', edad: 30, ciudad: 'Madrid', profesion: 'Desarrolladora' }
                        
                        console.log('¿Son el mismo objeto?', objeto1 === objeto2); // false

Exercise 4

Code:

                    
                        const colores = ['rojo', 'verde', 'azul', 'amarillo', 'naranja', 'morado'];
                        
                        // Destructuring con operador rest
                        const [primero, segundo, ...resto] = colores;

Example of usage:

                    
                        console.log('Primer elemento:', primero); // 'rojo'
                        console.log('Segundo elemento:', segundo); // 'verde'
                        console.log('Resto de elementos:', resto); 
                        // ['azul', 'amarillo', 'naranja', 'morado']
                        
                        console.log('Array original:', colores);
                        // ['rojo', 'verde', 'azul', 'amarillo', 'naranja', 'morado']

Exercise 5

Function:

                    
                        const saludar = (nombre, apellido, edad) => {
                            return `Hola, ${nombre} ${apellido}. Tienes ${edad} años.`;
                        };
                        
                        const datosPersona = ['Ana', 'García', 28];

Example of usage:

                    
                        // Llamada normal
                        const mensaje1 = saludar('Carlos', 'López', 35);
                        console.log(mensaje1); // Hola, Carlos López. Tienes 35 años.
                        
                        // Llamada usando spread
                        const mensaje2 = saludar(...datosPersona);
                        console.log(mensaje2); // Hola, Ana García. Tienes 28 años.

Exercise 6

Code:

                    
                        const objeto1 = {
                            nombre: 'Carlos',
                            edad: 30,
                            ciudad: 'Madrid'
                        };
                        
                        const objeto2 = {
                            profesion: 'Desarrollador',
                            email: 'carlos@email.com',
                            activo: true
                        };
                        
                        // Fusionando objetos con spread
                        const objetoFusionado = { ...objeto1, ...objeto2 };

Example of usage:

                    
                        console.log('Objeto 1:', objeto1);
                        // { nombre: 'Carlos', edad: 30, ciudad: 'Madrid' }
                        
                        console.log('Objeto 2:', objeto2);
                        // { profesion: 'Desarrollador', email: 'carlos@email.com', activo: true }
                        
                        console.log('Objeto fusionado:', objetoFusionado);
                        // { nombre: 'Carlos', edad: 30, ciudad: 'Madrid', 
                        //   profesion: 'Desarrollador', email: 'carlos@email.com', activo: true }
