6. Array Loops

⚠️ Important: Open the browser console to see the results

Exercise 1

Code:

                    
                        let names = ['Anna', 'Bernat', 'Clara'];
                        names.forEach(name => console.log(name));

Example:

                    
                        names.forEach(name => console.log(name));

Exercise 2

Code:

                    
                        let namesLoop = ['Anna', 'Bernat', 'Clara'];
                        for (const name of namesLoop) {
                            console.log(name);
                        }

Example:

                    
                        for (const name of namesLoop) {
                            console.log(name);
                        }

Exercise 3

Code:

                    
                        let numbers = [1, 2, 3, 4, 5, 6];
                        let evens = numbers.filter(num => num % 2 === 0);

Example:

                    
                        console.log('Original array:', numbers);
                        console.log('Even numbers:', evens);

Exercise 4

Code:

                    
                        let obj = { name: 'Ola', age: 25, city: 'Barcelona' };
                        for (const key in obj) {
                            console.log(key + ': ' + obj[key]);
                        }

Example:

                    
                        for (const key in obj) {
                            console.log(key + ': ' + obj[key]);
                        }

Exercise 5

Code:

                    
                        let numbersBreak = [1, 2, 3, 4, 5, 6];
                        for (const num of numbersBreak) {
                            if (num === 5) {
                                break;
                            }
                            console.log(num);
                        }

Example:

                    
                        for (const num of numbersBreak) {
                            if (num === 5) {
                                break;
                            }
                            console.log(num);
                        }

Exercise 6

Code:

                    
                        let namesIndex = ['Anna', 'Bernat', 'Clara'];
                        for (const [index, name] of namesIndex.entries()) {
                            console.log(index + ': ' + name);
                        }

Example:

                    
                        for (const [index, name] of namesIndex.entries()) {
                            console.log(index + ': ' + name);
                        }
